130 void FIPC_API::syncMotionRel(float iDist[],float iTimeSpeed, float iAccelTime){
131  // Primero verifica que todos los desplazamiento puedan realizarse
132  // y luego realiza la solicitud a cada eje
133  uint8_t i;
134  for (i = 0; i<AXIS_NUMBERS; i++) // check if can move that distance
135  if( (iDist[i])&&(!_axis[i]->canMoveRelative(iDist[i])) ) return;
136 
137  for (i = 0; i<AXIS_NUMBERS; i++) // check and config speeds
138  if( (iDist[i])&&(!_axis[i]->setSpeed(abs(iDist[i])/iTimeSpeed)) ) return;
139 
140  for (i = 0; i<AXIS_NUMBERS; i++) // check and config acceleration times
141  if( (iDist[i])&&(!_axis[i]->setAccelerationTime(iAccelTime)) ) return;
142 
143  for (i = 0; i<AXIS_NUMBERS; i++) // if all config were accepeted, then request action
144  if( iDist[i] ) _axis[i]->setAction(FIPC_Axis::ACTION_MOVE_RELATIVE,iDist[i]);
145 }
