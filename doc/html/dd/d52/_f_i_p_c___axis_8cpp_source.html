94 bool FIPC_Axis::setAction(uint8_t iAction, float iData){
95  switch(_axis_status) {
96  case STATUS_DISABLE:
97  if( iAction==ACTION_ENABLE ) {
98  _newExec = EXEC_ENABLE;
99  return true;
100  }
101  break;
102  case STATUS_NO_HOME:
103  if( iAction==ACTION_DISABLE ){
104  _newExec = EXEC_DISABLE;
105  return true;
106  }
107  if( iAction==ACTION_HOMING ){
108  _newExec = EXEC_HOMING;
109  return true;
110  }
111  break;
112  case STATUS_HOMING:
113  if( iAction==ACTION_STOP ){
114  _newExec = EXEC_STOP;
115  return true;
116  }
117  break;
118  case STATUS_READY:
119  if( iAction==ACTION_DISABLE ){
120  _newExec = EXEC_DISABLE;
121  return true;
122  }
123  if( ((iAction==ACTION_MOVE_RELATIVE)&&(FIPC_Axis::configMoveRelative(iData))) ||
124  ((iAction==ACTION_MOVE_ABSOLUTE)&&(FIPC_Axis::configMoveAbsolute(iData))) ){
125  _newExec = EXEC_RUN;
126  return true;
127  }
128  break;
129  case STATUS_MOVING:
130  if( iAction==ACTION_STOP ){
131  _newExec = EXEC_STOP;
132  return true;
133  }
134  break;
135  }
136  return false;
137 }

214 /*------------ PROCESO EN TIEMPO REAL ----------*/
215 void FIPC_Axis::exec(){
216  // 1° debe atender si se está desplazando
217  if( _axis_status==STATUS_MOVING ) {
218  if( _newExec==EXEC_STOP ) {
219  _Axis->stop();
220  _newExec = EXEC_WAIT;
221  }
222 
223  if( !_Axis->run() ) _axis_status = STATUS_READY;
224  return;
225  }
226 
227  // 2° Debe atender si se está buscando el cero
228  if( _axis_status==STATUS_HOMING ) {
229  if( _newExec==EXEC_STOP ) {
230  _Homing->stop();
231  _newExec = EXEC_WAIT;
232  _axis_status = STATUS_NO_HOME;
233  return;
234  }
235  if( !_Homing->run() ) _axis_status = STATUS_READY;
236  return;
237  }
238 
239  // 3° Espera por una acción de desplazamiento
240  if( _axis_status==STATUS_READY ) {
241  if( _newExec==EXEC_RUN ) {
242  _axis_status = STATUS_MOVING;
243  _newExec = EXEC_WAIT;
244  }
245  if( _newExec==EXEC_DISABLE ) {
246  _Axis->disableOutputs();
247  _Homing->stop();
248  _axis_status = STATUS_DISABLE;
249  _newExec = EXEC_WAIT;
250  }
251  return;
252  }
253 
254  // 4° espera para buscar el cero
255  if( _axis_status==STATUS_NO_HOME ) {
256  if( _newExec==EXEC_HOMING ) {
257  _axis_status = STATUS_HOMING;
258  _newExec = EXEC_WAIT;
259  }
260  if( _newExec==EXEC_DISABLE ) {
261  _Axis->disableOutputs();
262  _Homing->stop();
263  _axis_status = STATUS_DISABLE;
264  _newExec = EXEC_WAIT;
265  }
266  return;
267  }
268 
269  // 5° espera para ser habilitado
270  if( _axis_status==STATUS_DISABLE ) {
271  if( _newExec==EXEC_ENABLE ) {
272  _Axis->enableOutputs();
273  _axis_status = STATUS_NO_HOME;
274  _newExec = EXEC_WAIT;
275  }
276  return;
277  }
278 }
